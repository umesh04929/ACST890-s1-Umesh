
#!/bin/bash
# save as triplet-stop-altered.sh
# splits a sequence into triplets, stops when 'taa' ecnountered, read at least 9 characters in every run
x=0
while [ -n "${1:$x:3}" ]; do
  seq=$seq${1:$x:3}" "
  x=$(expr $x + 3)
  if [ "${1:$x:3}" == taa ] && [ "$x" -gt 8 ]; then
    break
  fi
done
echo "$seq"
